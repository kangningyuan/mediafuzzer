# fuzzing 模块开发文档

## 1. 模块职责

将 LibFuzzer 与 Qiling 模拟执行深度集成，实现：
- 基于 LibFuzzer 的变异引擎驱动
- 覆盖率引导（基本块级）
- 文件格式感知变异（保持格式骨架不变）
- 单函数模糊测试任务控制

## 2. 文件结构

```
src/fuzzing/
├── __init__.py        # 导出 FuzzWorker, run_fuzzing
├── harness.py         # LibFuzzer 测试驱动生成与编译
├── coverage.py        # 基本块覆盖率收集与反馈
├── format_aware.py    # 文件格式感知变异
├── mutator.py         # 自定义突变策略
└── fuzz_worker.py     # 单个函数的模糊测试任务控制
```

## 4. 详细接口规格

### 4.1 harness.py

**核心挑战**：LibFuzzer 是 C 语言框架，我们的目标函数运行在 Qiling（Python）模拟环境中。需要桥接两者。

**方案**：编译一个 C harness 共享库，由 Python 通过 ctypes 加载。harness 的 `LLVMFuzzerTestOneInput` 通过回调函数将数据传递给 Python 侧的 Qiling 执行。

```python
def generate_harness_source(
    func_name: str,
    callback_symbol: str = "python_fuzz_callback",
) -> str:
    """
    生成 LibFuzzer harness C 源码。

    生成的 harness 结构：
    1. 声明外部回调函数指针
    2. 实现 LLVMFuzzerTestOneInput
    3. 实现 LLVMFuzzerCustomMutator（格式感知变异入口）
    4. 声明 __libfuzzer_extra_counters（覆盖率位图）

    Args:
        func_name: 目标函数名（用于日志和标识）
        callback_symbol: Python 回调函数的符号名

    Returns:
        C 源码字符串
    """
    return f"""
// Auto-generated by MediaFuzzer for: {func_name}

#include <stdint.h>
#include <stddef.h>
#include <string.h>

// Python 回调函数类型: int callback(const uint8_t *data, size_t size)
// 返回值: 0=正常, -1=检测到异常
typedef int (*fuzz_callback_t)(const uint8_t *, size_t);
static fuzz_callback_t g_fuzz_callback = NULL;

// 覆盖率位图（与 Qiling 侧共享）
#define COV_BITMAP_SIZE {settings.COV_BITMAP_SIZE}
uint8_t __libfuzzer_extra_counters[COV_BITMAP_SIZE];
static int cov_bitmap_dirty = 0;

// 设置回调函数（由 Python ctypes 调用）
void set_fuzz_callback(fuzz_callback_t cb) {{
    g_fuzz_callback = cb;
}}

// 重置覆盖率位图（每轮变异前由 Python 调用）
void reset_coverage_bitmap(void) {{
    memset(__libfuzzer_extra_counters, 0, COV_BITMAP_SIZE);
    cov_bitmap_dirty = 0;
}}

// LibFuzzer 入口
int LLVMFuzzerTestOneInput(const uint8_t *data, size_t size) {{
    if (!g_fuzz_callback) return 0;
    if (size < 1) return 0;

    // 重置位图
    reset_coverage_bitmap();

    // 调用 Python 侧的 Qiling 执行
    int result = g_fuzz_callback(data, size);

    return result;
}}

// 自定义变异器（格式感知变异的入口）
// 如果格式感知模块可用，使用格式感知变异
// 否则回退到 LibFuzzer 默认变异
extern size_t mediafuzzer_custom_mutator(
    uint8_t *data, size_t size, size_t max_size, unsigned int seed);

size_t LLVMFuzzerCustomMutator(
    uint8_t *data, size_t size, size_t max_size, unsigned int seed) {{
    return mediafuzzer_custom_mutator(data, size, max_size, seed);
}}
"""
```

```python
def compile_harness(
    harness_source: str,
    output_path: str | None = None,
) -> str:
    """
    编译 harness 为共享库。

    编译命令：
    {CLANG_PATH} -shared -fPIC -fsanitize=fuzzer-no-link \
        -o {output_path} harness.c

    注意：使用 -fsanitize=fuzzer-no-link 而非 -fsanitize=fuzzer，
    因为 LibFuzzer 主循环由 Python 侧驱动。

    Args:
        harness_source: C 源码
        output_path: 输出 .so 路径

    Returns:
        编译产物 .so 的绝对路径

    Raises:
        RuntimeError: 编译失败
    """
    import tempfile
    if output_path is None:
        output_path = os.path.join(settings.OUTPUT_BASE_DIR, "harness", "libfuzzer_harness.so")

    with tempfile.NamedTemporaryFile(suffix=".c", mode="w", delete=False) as f:
        f.write(harness_source)
        src_path = f.name

    try:
        cmd = [
            settings.CLANG_PATH,
            "-shared", "-fPIC",
            "-fsanitize=fuzzer-no-link",
            "-o", output_path,
            src_path,
        ]
        result = subprocess.run(cmd, capture_output=True, text=True)
        if result.returncode != 0:
            raise RuntimeError(f"Harness compilation failed: {result.stderr}")
    finally:
        os.unlink(src_path)

    return output_path
```

```python
class HarnessBridge:
    """
    Python 侧的 LibFuzzer 桥接。

    通过 ctypes 加载编译后的 harness .so，
    并提供 Python 回调给 Qiling 执行。
    """

    def __init__(self, harness_so_path: str, emulated_func: 'EmulatedJNIFunc'):
        self.lib = ctypes.CDLL(harness_so_path)
        self.emulated_func = emulated_func

        # 设置回调函数类型
        CALLBACK_TYPE = ctypes.CFUNCTYPE(ctypes.c_int, ctypes.POINTER(ctypes.c_uint8), ctypes.c_size_t)
        self._callback = CALLBACK_TYPE(self._fuzz_callback)
        self.lib.set_fuzz_callback(self._callback)

        # 获取覆盖率位图引用
        self.cov_bitmap = (ctypes.c_uint8 * settings.COV_BITMAP_SIZE)()
        # 将位图地址传给 harness（通过 ctypes 直接访问共享库的全局变量）

    def _fuzz_callback(self, data_ptr, size) -> int:
        """
        LibFuzzer 变异后的回调，执行 Qiling 模拟。

        这个函数被 LibFuzzer 的变异循环在每次迭代时调用。
        """
        # 读取变异数据
        input_data = bytes(ctypes.string_at(data_ptr, size))

        try:
            result = self.emulated_func.call_function(input_data)
            return 0  # 正常执行
        except Exception as e:
            logging.debug(f"Emulation error: {e}")
            return -1  # 检测到异常

    def reset_coverage(self) -> None:
        """重置覆盖率位图"""
        self.lib.reset_coverage_bitmap()

    def get_coverage_bitmap(self) -> bytes:
        """获取当前覆盖率位图快照"""
        return bytes(self.cov_bitmap)
```

### 4.2 coverage.py

```python
class CoverageTracker:
    """
    基本块覆盖率追踪器。

    在 Qiling 模拟执行的每个基本块入口处，计算地址哈希
    并更新覆盖率位图。位图与 LibFuzzer 共享，实现覆盖率引导。

    位图索引计算：
    index = (prev_block_hash ^ current_block_hash) % COV_BITMAP_SIZE
    这捕获了边（edge）覆盖，而非仅点（block）覆盖。
    """

    def __init__(self, bitmap_size: int | None = None):
        self.bitmap_size = bitmap_size or settings.COV_BITMAP_SIZE
        self.bitmap = bytearray(self.bitmap_size)
        self._prev_hash: int = 0
        self._total_blocks: int = 0
        self._covered_edges: set[int] = set()

    def on_basic_block(self, ql, addr: int, size: int) -> None:
        """
        基本块入口钩子。由 Qiling 的 UC_HOOK_CODE 触发。

        Args:
            ql: Qiling 实例
            addr: 基本块起始地址
            size: 基本块大小（字节）
        """
        # 计算当前块哈希（使用地址的简单哈希）
        curr_hash = (addr >> 4) ^ (addr << 8)
        curr_hash &= 0xFFFFFFFF

        # 计算边索引
        edge_idx = (self._prev_hash ^ curr_hash) % self.bitmap_size

        # 更新位图（AFL 风格：记录命中次数）
        if self.bitmap[edge_idx] < 255:
            self.bitmap[edge_idx] += 1

        # 记录新边
        self._covered_edges.add(edge_idx)
        self._total_blocks += 1

        # 更新前驱哈希
        self._prev_hash = curr_hash >> 1  # 类似 AFL 的 ROR

    def reset(self) -> None:
        """重置位图（每轮模糊测试前调用）"""
        self.bitmap = bytearray(self.bitmap_size)
        self._prev_hash = 0
        self._total_blocks = 0

    def get_new_edges(self, previous_bitmap: bytes) -> set[int]:
        """
        对比上一轮位图，返回本轮新覆盖的边。

        用于判断变异是否产生新覆盖，决定是否保留该输入。
        """
        new_edges = set()
        for i in range(self.bitmap_size):
            if self.bitmap[i] > 0 and previous_bitmap[i] == 0:
                new_edges.add(i)
        return new_edges

    @property
    def coverage_ratio(self) -> float:
        """当前覆盖率（已覆盖边 / 总位图大小）"""
        covered = sum(1 for b in self.bitmap if b > 0)
        return covered / self.bitmap_size

    def register_hooks(self, emulated_func: 'EmulatedJNIFunc') -> None:
        """
        在 EmulatedJNIFunc 的 Qiling 实例上注册覆盖率钩子。

        使用 Qiling 的 hook_code：
        ql.hook_code(self.on_basic_block)
        """
        if emulated_func.ql is None:
            raise RuntimeError("EmulatedJNIFunc not initialized")

        emulated_func.ql.hook_code(self.on_basic_block)
```

### 4.3 format_aware.py

```python
class FormatAwareMutator:
    """
    文件格式感知变异器。

    根据 LLM 识别的文件格式，加载对应骨架定义，
    在变异时保持格式关键字段不变，只对数据内容变异。

    实现 LLVMFuzzerCustomMutator 逻辑。
    """

    def __init__(self, format_name: str | None = None):
        """
        Args:
            format_name: 文件格式名（如 "GIF", "JPEG"），
                        None 表示不使用格式感知（回退到默认变异）
        """
        self.format_name = format_name
        self.skeleton: FormatSkeleton | None = None

        if format_name:
            try:
                from config.file_formats import get_format
                self.skeleton = get_format(format_name)
            except KeyError:
                logging.warning(f"Format '{format_name}' not registered, using raw mutation")
                self.skeleton = None

    def mutate(self, data: bytearray, max_size: int, seed: int) -> bytearray:
        """
        格式感知变异。

        策略：
        1. 如果有骨架定义，识别 fixed 字段的偏移和大小
        2. 对非 fixed 区域执行标准变异操作：
           - 随机字节替换
           - 字节插入/删除
           - 块复制
           - 位翻转
           - 算术操作
        3. 恢复所有 fixed 字段为骨架默认值
        4. 如果数据超出 max_size，截断非 fixed 区域

        Args:
            data: 当前输入数据
            max_size: 最大允许大小
            seed: 随机种子

        Returns:
            变异后的数据
        """
        if self.skeleton is None:
            return self._raw_mutate(data, max_size, seed)

        rng = random.Random(seed)

        # 1. 执行标准变异
        mutated = self._raw_mutate(data, max_size, seed)

        # 2. 恢复 fixed 字段
        if self.skeleton and len(mutated) >= self.skeleton.min_seed_size:
            for field in self.skeleton.fields:
                if field.fixed and field.offset + field.size <= len(mutated):
                    mutated[field.offset:field.offset + field.size] = field.default_value[:field.size]

        # 3. 验证魔数字
        if self.skeleton and len(mutated) >= len(self.skeleton.magic):
            mutated[:len(self.skeleton.magic)] = self.skeleton.magic

        return mutated

    def _raw_mutate(self, data: bytearray, max_size: int, seed: int) -> bytearray:
        """
        标准变异操作（无格式约束）。

        操作列表：
        1. 随机字节替换 (probability 0.4)
        2. 位翻转 (probability 0.15)
        3. 算术增减 (probability 0.15)
        4. 块复制 (probability 0.1)
        5. 字节插入 (probability 0.1)
        6. 字节删除 (probability 0.05)
        7. 交叉覆盖 (probability 0.05)
        """
        rng = random.Random(seed)
        result = bytearray(data)

        if len(result) == 0:
            # 空输入：生成随机数据
            size = rng.randint(1, min(64, max_size))
            return bytearray(rng.getrandbits(8) for _ in range(size))

        op = rng.random()

        if op < 0.4:
            # 随机字节替换
            num_replacements = rng.randint(1, max(1, len(result) // 10))
            for _ in range(num_replacements):
                pos = rng.randint(0, len(result) - 1)
                result[pos] = rng.getrandbits(8)

        elif op < 0.55:
            # 位翻转
            pos = rng.randint(0, len(result) - 1)
            bit = 1 << rng.randint(0, 7)
            result[pos] ^= bit

        elif op < 0.7:
            # 算术增减
            pos = rng.randint(0, len(result) - 1)
            delta = rng.choice([-35, -1, 1, 35])
            result[pos] = (result[pos] + delta) & 0xFF

        elif op < 0.8:
            # 块复制
            if len(result) > 2:
                src = rng.randint(0, len(result) - 1)
                dst = rng.randint(0, len(result) - 1)
                length = rng.randint(1, min(16, len(result) - max(src, dst)))
                for i in range(length):
                    if dst + i < len(result):
                        result[dst + i] = result[src + i] if src + i < len(result) else 0

        elif op < 0.9:
            # 字节插入
            if len(result) < max_size:
                pos = rng.randint(0, len(result))
                num_bytes = rng.randint(1, min(4, max_size - len(result)))
                new_bytes = bytearray(rng.getrandbits(8) for _ in range(num_bytes))
                result = result[:pos] + new_bytes + result[pos:]

        elif op < 0.95:
            # 字节删除
            if len(result) > 1:
                pos = rng.randint(0, len(result) - 1)
                num_bytes = rng.randint(1, min(4, len(result) - 1))
                result = result[:pos] + result[pos + num_bytes:]

        else:
            # 交叉覆盖（使用另一个已知种子的片段）
            # 简化实现：从数据中随机选一段覆盖到另一位置
            if len(result) > 4:
                src = rng.randint(0, len(result) // 2)
                dst = rng.randint(len(result) // 2, len(result) - 1)
                length = rng.randint(1, min(8, len(result) - max(src, dst)))
                for i in range(length):
                    if dst + i < len(result) and src + i < len(result):
                        result[dst + i] = result[src + i]

        # 截断到 max_size
        if len(result) > max_size:
            result = result[:max_size]

        return result

    def generate_seed(self) -> bytes:
        """根据格式骨架生成初始种子"""
        if self.skeleton:
            return self.skeleton.generate_seed()
        # 默认种子：随机 64 字节
        return bytes(random.getrandbits(8) for _ in range(64))
```

### 4.4 mutator.py

```python
# 全局自定义变异器函数（由 harness.c 的 LLVMFuzzerCustomMutator 调用）
# 这是一个 C 可调用的函数，通过 ctypes 注册到 harness 共享库

class CustomMutatorBridge:
    """
    将 Python 侧的格式感知变异器桥接到 C 侧的
    LLVMFuzzerCustomMutator。
    """

    def __init__(self, format_aware: FormatAwareMutator, harness_lib):
        self.format_aware = format_aware
        self.harness_lib = harness_lib

        # 注册 mediafuzzer_custom_mutator 实现
        MUTATOR_FUNC_TYPE = ctypes.CFUNCTYPE(
            ctypes.c_size_t,           # 返回值：变异后大小
            ctypes.POINTER(ctypes.c_uint8),  # data
            ctypes.c_size_t,           # size
            ctypes.c_size_t,           # max_size
            ctypes.c_uint,             # seed
        )
        self._mutator_func = MUTATOR_FUNC_TYPE(self._custom_mutator)
        # 通过 ctypes 将函数指针注册到共享库

    def _custom_mutator(self, data_ptr, size, max_size, seed) -> int:
        """
        LLVMFuzzerCustomMutator 的 Python 实现。

        被 harness.c 中的 LLVMFuzzerCustomMutator 调用。
        """
        # 读取当前数据
        data = bytearray(ctypes.string_at(data_ptr, size))

        # 执行格式感知变异
        mutated = self.format_aware.mutate(data, max_size, seed)

        # 写回缓冲区
        write_size = min(len(mutated), max_size)
        ctypes.memmove(data_ptr, bytes(mutated[:write_size]), write_size)

        return write_size
```

### 4.5 fuzz_worker.py

```python
@dataclass
class FuzzResult:
    """单个函数的模糊测试结果"""
    func_sig: JNISignature           # 被测函数签名
    total_runs: int                  # 总变异轮次
    total_time: float                # 总运行时间（秒）
    crashes: list[dict]              # 崩溃列表
    memory_errors: list[dict]        # 内存错误列表
    coverage_ratio: float            # 最终覆盖率
    unique_crashes: int              # 去重后崩溃数
    seed_corpus_dir: str             # 种子语料目录
    output_dir: str                  # 输出目录

class FuzzWorker:
    """
    单个函数的模糊测试任务控制器。

    协调 harness 编译、Qiling 环境初始化、覆盖率追踪、
    格式感知变异和结果收集。
    """

    def __init__(self, func_info: MultimediaFuncInfo, output_dir: str):
        """
        Args:
            func_info: LLM 筛选出的多媒体函数信息
            output_dir: 本任务的输出目录
        """
        self.func_info = func_info
        self.output_dir = output_dir
        self.emulated_func: EmulatedJNIFunc | None = None
        self.cov_tracker: CoverageTracker | None = None
        self.format_mutator: FormatAwareMutator | None = None
        self.harness_bridge: HarnessBridge | None = None
        self._crashes: list[dict] = []
        self._memory_errors: list[dict] = []
        self._stop_requested = False

    def setup(self) -> None:
        """
        初始化模糊测试环境。

        Steps:
        1. 创建 EmulatedJNIFunc 并初始化
        2. 创建 CoverageTracker 并注册钩子
        3. 创建 FormatAwareMutator（根据 LLM 识别的格式）
        4. 生成并编译 harness
        5. 创建 HarnessBridge
        6. 生成初始种子语料
        """
        os.makedirs(self.output_dir, exist_ok=True)
        os.makedirs(os.path.join(self.output_dir, "seeds"), exist_ok=True)
        os.makedirs(os.path.join(self.output_dir, "crashes"), exist_ok=True)

        # 1. 初始化模拟环境
        self.emulated_func = EmulatedJNIFunc(
            so_path=self.func_info.jni_signature.so_path,
            func_symbol=self.func_info.jni_signature.native_symbol,
        )
        self.emulated_func.initialize()

        # 2. 覆盖率追踪
        self.cov_tracker = CoverageTracker()
        self.cov_tracker.register_hooks(self.emulated_func)

        # 3. 格式感知变异器
        self.format_mutator = FormatAwareMutator(
            format_name=self.func_info.file_format
            if self.func_info.file_format != "UNKNOWN"
            else None
        )

        # 4. 编译 harness
        harness_src = generate_harness_source(self.func_info.jni_signature.native_symbol)
        harness_so = compile_harness(harness_src)

        # 5. 桥接
        self.harness_bridge = HarnessBridge(harness_so, self.emulated_func)

        # 6. 生成初始种子
        seed = self.format_mutator.generate_seed()
        seed_path = os.path.join(self.output_dir, "seeds", "seed_0")
        with open(seed_path, "wb") as f:
            f.write(seed)

    def run(self, max_runs: int | None = None, timeout: int | None = None) -> FuzzResult:
        """
        执行模糊测试主循环。

        Args:
            max_runs: 最大变异轮次
            timeout: 总超时秒数

        Returns:
            FuzzResult 包含所有发现

        Implementation:
        本实现采用纯 Python 驱动的变异循环，而非依赖 LibFuzzer 的
        内部主循环。原因是 Qiling 模拟执行在 Python 侧，与
        LibFuzzer 的 C 主循环交互复杂。

        循环逻辑：
        1. 加载种子语料
        2. 循环迭代：
           a. 从语料中选择一个输入
           b. 执行格式感知变异
           c. 重置覆盖率位图
           d. 调用 EmulatedJNIFunc.call_function(mutated_data)
           e. 检查覆盖率是否有新边
           f. 如果有新覆盖，将变异输入加入语料
           g. 检查是否崩溃或内存异常
           h. 记录结果
        3. 达到 max_runs 或 timeout 后退出
        """
        max_runs = max_runs or settings.LIBFUZZER_MAX_RUNS
        timeout = timeout or settings.LIBFUZZER_TIMEOUT
        start_time = time.time()

        # 加载种子
        corpus: list[bytes] = []
        seeds_dir = os.path.join(self.output_dir, "seeds")
        for fname in os.listdir(seeds_dir):
            with open(os.path.join(seeds_dir, fname), "rb") as f:
                corpus.append(f.read())

        if not corpus:
            corpus.append(self.format_mutator.generate_seed())

        run_count = 0
        seed = 0

        while run_count < max_runs and not self._stop_requested:
            # 检查超时
            if time.time() - start_time > timeout:
                logging.info(f"Fuzzing timeout after {run_count} runs")
                break

            # 从语料中选择输入
            idx = seed % len(corpus)
            current_input = bytearray(corpus[idx])

            # 变异
            mutated = self.format_mutator.mutate(current_input, max_size=1048576, seed=seed)

            # 保存前一轮覆盖率位图
            prev_bitmap = bytes(self.cov_tracker.bitmap)

            # 重置覆盖率
            self.cov_tracker.reset()

            # 执行
            try:
                self.emulated_func.call_function(bytes(mutated))
                exec_ok = True
            except Exception as e:
                exec_ok = False
                self._record_crash(mutated, str(e), run_count)

            # 检查新覆盖
            new_edges = self.cov_tracker.get_new_edges(prev_bitmap)
            if new_edges and exec_ok:
                # 有新覆盖，加入语料
                corpus.append(bytes(mutated))
                # 保存到种子目录
                seed_path = os.path.join(seeds_dir, f"seed_{len(corpus)}")
                with open(seed_path, "wb") as f:
                    f.write(mutated)

            run_count += 1
            seed += 1

        return FuzzResult(
            func_sig=self.func_info.jni_signature,
            total_runs=run_count,
            total_time=time.time() - start_time,
            crashes=self._crashes,
            memory_errors=self._memory_errors,
            coverage_ratio=self.cov_tracker.coverage_ratio,
            unique_crashes=len({c["crash_hash"] for c in self._crashes}),
            seed_corpus_dir=seeds_dir,
            output_dir=self.output_dir,
        )

    def _record_crash(self, input_data: bytes, error_msg: str, run_num: int) -> None:
        """记录崩溃信息"""
        crash_hash = hashlib.md5(input_data).hexdigest()[:8]
        crash_info = {
            "crash_hash": crash_hash,
            "input_data": input_data.hex(),
            "error_message": error_msg,
            "run_number": run_num,
            "timestamp": time.time(),
        }
        self._crashes.append(crash_info)

        # 保存崩溃输入到文件
        crash_path = os.path.join(self.output_dir, "crashes", f"crash_{crash_hash}")
        with open(crash_path, "wb") as f:
            f.write(input_data)

    def stop(self) -> None:
        """请求停止模糊测试"""
        self._stop_requested = True

    def teardown(self) -> None:
        """清理资源"""
        if self.emulated_func:
            self.emulated_func.destroy()
```

## 5. 关键实现细节

### 5.1 LibFuzzer 集成策略选择

**方案A（推荐用于M3-M4）**：纯 Python 驱动变异循环
- 优点：与 Qiling 集成简单，调试方便
- 缺点：无法利用 LibFuzzer 的高效内部变异引擎
- 适用：原型验证阶段

**方案B（目标方案）**：LibFuzzer 主循环 + ctypes 回调
- 优点：利用 LibFuzzer 的成熟变异和语料管理
- 缺点：Python/C 桥接复杂，覆盖率反馈需要共享内存
- 适用：性能优化阶段

**方案C（最终方案）**：`__libfuzzer_extra_counters` 共享覆盖率
- 通过 `__libfuzzer_extra_counters` 全局数组将 Qiling 侧的覆盖率反馈给 LibFuzzer
- LibFuzzer 自动读取该数组用于覆盖率引导决策
- 需要确保 Qiling 侧的覆盖率位图写入与 LibFuzzer 读取同步

### 5.2 覆盖率反馈机制

```
Qiling 模拟执行                     LibFuzzer
       │                               │
       ▼                               ▼
  基本块钩子                     __libfuzzer_extra_counters[]
       │                               │
       ▼                               ▼
  哈希计算 ──────写入──────▶  共享位图  ───读取──▶ 覆盖率引导决策
```

### 5.3 崩溃去重

使用输入数据的 MD5 哈希前8位作为崩溃指纹。相同指纹的崩溃视为同一漏洞的重复触发。

## 7. 错误处理

| 异常场景 | 处理方式 |
|---------|---------|
| harness 编译失败 | raise RuntimeError，中止该函数的模糊测试 |
| Qiling 初始化失败 | raise，中止该函数 |
| 模拟执行崩溃 | 记录为崩溃，继续下一轮 |
| 模拟执行超时 | 记录为超时，继续下一轮 |
| 语料为空 | 使用格式感知种子 |
| 内存不足 | 记录错误，停止模糊测试 |
